\question \textbf{Landau Symbols}

\begin{parts}
\part Show the following: \(\forall k, l \in \mathbb{Z}, k > l : n^l = o(n^k)\)


\begin{solution}

\textbf{Def.:} \(o(f) := \{g: \mathbb{N} \rightarrow \mathbb{R}_+: \lim\limits_{n \to \infty}\frac{g(n)}{f(n)}=0\}\)

Let \(f(n)=n^k\), \(g(n)=n^l\).

\(\lim\limits_{n \to \infty}\frac{g(n)}{f(n)}=\lim\limits_{n \to \infty}\frac{n^l}{n^k}=\lim\limits_{n \to \infty}n^{l-k}=\lim\limits_{n \to \infty}n^{-(-l+k)}=\lim\limits_{n \to \infty}\frac{1}{n^{k-l}} \rightarrow 0\) since \(n^{k-l} > 1\)

\end{solution}
